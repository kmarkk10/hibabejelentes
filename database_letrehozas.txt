CREATE DATABASE IF NOT EXISTS hibajegyek
CHARACTER SET utf8mb4
COLLATE utf8mb4_hungarian_ci;

USE hibajegyek;

CREATE TABLE IF NOT EXISTS admin (
    id INT AUTO_INCREMENT PRIMARY KEY,
    felhasznalo_nev VARCHAR(50) NOT NULL UNIQUE,
    jelszo VARCHAR(255) NOT NULL,
    pozicio VARCHAR(50) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS tanar (
    id INT AUTO_INCREMENT PRIMARY KEY,
    felhasznalo_nev VARCHAR(50) NOT NULL UNIQUE,
    jelszo VARCHAR(255) NOT NULL,
    pozicio VARCHAR(50) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE IF NOT EXISTS hibak (
    hiba_kod INT AUTO_INCREMENT PRIMARY KEY,
    hiba_cim VARCHAR(255) NOT NULL,
    hiba TEXT NOT NULL,
    datum DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    statusz VARCHAR(50) NOT NULL DEFAULT 'Kezelésre vár',
    tanar_nev VARCHAR(100) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
